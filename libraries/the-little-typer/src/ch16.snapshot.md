# 16. If It's All the Same to You

## `zero_dec`

``` cicada
import { Dec, use_nat_equal } from "ch15.snapshot.md"

def zero_dec(n: Nat): Dec(Equal(Nat, zero, n)) =

  let motive(target: Nat): Type = Dec(Equal(Nat, zero, target))

  let base: Dec(Equal(Nat, zero, zero)) = inl(same)

  let step(
    prev: Nat,
    almost: Dec(Equal(Nat, zero, prev)),
  ): Dec(Equal(Nat, zero, add1(prev))) =
    inr((equal) => use_nat_equal(zero, add1(prev), equal))

  nat_ind(n, motive, base, step)

zero_dec(0)
zero_dec(1)
zero_dec(2)
zero_dec(3)
```
